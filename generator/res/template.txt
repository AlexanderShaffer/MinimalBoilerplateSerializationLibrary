export module mbsl;
import std;

namespace {
template<typename T>
concept Number = std::is_arithmetic_v<T> || std::is_enum_v<T>;

template<typename T>
concept StdArray = std::same_as<T, std::array<typename T::value_type, sizeof(T) / sizeof(typename T::value_type)>>;

template<typename T>
concept EndiannessResistantNumber = Number<T> && sizeof(T) == 1;

template<typename T>
concept EndiannessResistantStdArray = StdArray<T> && EndiannessResistantNumber<typename T::value_type>;

template<typename T>
concept EndiannessResistant = EndiannessResistantNumber<T> || EndiannessResistantStdArray<T>;

template<typename T>
concept EndiannessSusceptibleNumber = Number<T> && sizeof(T) > 1;

template<typename T>
concept EndiannessSusceptibleStdArray = StdArray<T> && EndiannessSusceptibleNumber<typename T::value_type>;

template<typename T>
concept EndiannessSusceptible = EndiannessSusceptibleNumber<T> || EndiannessSusceptibleStdArray<T>;

template<typename T>
concept Noncontiguous = false; // TODO: Implement this concept

template<typename T>
consteval void assert_supported() {
  static_assert(EndiannessResistant<T> || EndiannessSusceptible<T> || Noncontiguous<T>, "Unsupported member type");
}

template<typename T>
consteval bool is_valid_member_order() {
  assert_supported<T>();
  return true;
}

template<typename CurrentT, typename NextT, typename... Ts>
consteval bool is_valid_member_order() {
  assert_supported<CurrentT>();
  return false;
}

template<Noncontiguous CurrentT, typename NextT, typename... Ts>
consteval bool is_valid_member_order() {
  assert_supported<NextT>();
  static_assert(!EndiannessResistant<NextT>, "Endianness resistant members must precede noncontiguous members");
  static_assert(!EndiannessSusceptible<NextT>, "Endianness susceptible members must precede noncontiguous members");
  return is_valid_member_order<NextT, Ts...>();
}

template<EndiannessSusceptible CurrentT, typename NextT, typename... Ts>
consteval bool is_valid_member_order() {
  assert_supported<NextT>();
  static_assert(!EndiannessResistant<NextT>, "Endianness resistant members must precede endianness susceptible members");
  return is_valid_member_order<NextT, Ts...>();
}

template<EndiannessResistant CurrentT, typename NextT, typename... Ts>
consteval bool is_valid_member_order() {
  assert_supported<NextT>();
  return is_valid_member_order<NextT, Ts...>();
}
} // namespace
